var DB_OPEN = 'fsHandlesDB';
        var DB_OBJ_STORE = 'handles';
        var RW_PERM = 'readwrite';
        var RO_PERM = 'readonly';
        var RW_MODE = { mode: RW_PERM };
        var R_PERM = 'read';
        // Usage
        const KEY = 'local1-dir';

        function openDB() {
            return new Promise((resolve, reject) => {
                const req = indexedDB.open(DB_OPEN, 1);
                req.onupgradeneeded = () => {
                    if (!req.result.objectStoreNames.contains(DB_OBJ_STORE)) {
                        req.result.createObjectStore(DB_OBJ_STORE);
                    }
                };
                req.onerror = () => reject(req.error);
                req.onsuccess = () => resolve(req.result);
            });
        }

        async function indexdb_put(obj, key) {
            const db = await openDB();
            return new Promise((resolve, reject) => {
                const tx = db.transaction(DB_OBJ_STORE, RW_PERM);
                tx.oncomplete = () => resolve();
                tx.onerror = () => reject(tx.error);
                tx.objectStore(DB_OBJ_STORE).put(obj, key);
            });
        }

        async function indexdb_get(key) {
            const db = await openDB();
            return new Promise((resolve, reject) => {
                const tx = db.transaction(DB_OBJ_STORE, RO_PERM);
                const req = tx.objectStore(DB_OBJ_STORE).get(key);
                req.onsuccess = () => resolve(req.result ?? null);
                req.onerror = () => reject(req.error);
            });
        }

        async function verifyPermission(handle) {
            res = await handle.requestPermission(RW_MODE);
            //alert("request permission: " + res);

            if (res === 'granted') {
                return true;
            }
            return false;
        }

        async function restoreOrPickDir() {

            try {
                // 1) Try saved handle first
                let dir = await indexdb_get(KEY);

                if (!await verifyPermission(dir)) {
                    dir = null;
                }

                if (!dir) {
                    dir = await window.showDirectoryPicker(RW_MODE);
                }
                await indexdb_put(dir, KEY);

                return dir;
            }
            catch (e) {
                console.warn('Directory access failed:', e);
                alert('Directory access failed. Please try again.');
                return null;
            }
        }

        async function listFiles(dirHandle) {
            const files = [];
            for await (const entry of dirHandle.values()) {
                if (entry.kind === 'file') {
                    files.push(entry); // FileSystemFileHandle
                }
            }
            // Sort by name for stable UX
            files.sort((a, b) => a.name.localeCompare(b.name));
            return files;
        }

        const ulFiles = document.getElementById('file-list');       // <ul> to render file links

        let pendingDirHandle = null;
        (async function autoRestore() {
            const dir = await indexdb_get(KEY);
            if (!dir) return;

            try {
                const state = await dir.queryPermission(RW_MODE);
                if (state === 'granted') {
                    await renderFileList(dir);
                } else {
                    pendingDirHandle = dir;
                    showReconnectOverlay();
                    return;
                }
            } catch (e) {
                console.warn('Auto-restore failed:', e);
            }
        })();

        /* ===== PINNED FILES — localStorage ===== */
        const PF_KEY = 'toolbox-pinned-files';

        function getPinnedFiles() {
            try { return JSON.parse(localStorage.getItem(PF_KEY) || '[]'); } catch { return []; }
        }

        function setPinnedFiles(names) {
            localStorage.setItem(PF_KEY, JSON.stringify(names));
        }

        function isPinned(name) {
            return getPinnedFiles().includes(name);
        }

        function togglePin(name) {
            const pinned = getPinnedFiles();
            const idx = pinned.indexOf(name);
            if (idx === -1) pinned.push(name);
            else pinned.splice(idx, 1);
            setPinnedFiles(pinned);
        }

        // Render pinned file links into the header area
        // Must be called after renderFileList so _fileHandleMap is populated
        const _fileHandleMap = new Map(); // name → FileSystemFileHandle

        function renderPinnedHeader() {
            const container = document.getElementById('pinned-files-container');
            if (!container) return;
            container.innerHTML = '';

            const pinned = getPinnedFiles();
            pinned.forEach((name, i) => {
                const handle = _fileHandleMap.get(name);
                if (!handle) return; // directory not loaded yet

                const a = document.createElement('a');
                a.className = 'fillLink';
                a.href = '#';
                a.textContent = name;
                a.addEventListener('click', async (e) => {
                    e.preventDefault();
                    // trigger same open-in-editor flow as the file list link
                    document.querySelector(`#file-list a[data-filename="${CSS.escape(name)}"]`)?.click();
                });
                container.appendChild(a);
                if (i < pinned.length - 1) container.appendChild(document.createTextNode(' | '));
            });
        }

        // Render file list as clickable links
        async function renderFileList(dirHandle) {
            const editor = document.getElementById('editor');

            const entries = await listFiles(dirHandle);
            ulFiles.innerHTML = '';
            _fileHandleMap.clear();
            let inputAbortController = null;

            for (const handle of entries) {
                _fileHandleMap.set(handle.name, handle);

                // Auto-populate any section whose id matches the filename
                const eleSection = document.getElementById(handle.name);
                if (eleSection) {
                    const file = await handle.getFile();
                    const text = await file.text();
                    if (handle.name === 'json2.json') {
                        renderLinks(text);
                    } else {
                        eleSection.style.whiteSpace = 'pre-wrap';
                        eleSection.textContent = text;
                    }
                }

                const li = document.createElement('li');

                // Pin button
                const pinBtn = document.createElement('button');
                pinBtn.className = 'file-pin-btn' + (isPinned(handle.name) ? ' pinned' : '');
                pinBtn.title = isPinned(handle.name) ? 'Unpin from header' : 'Pin to header';
                pinBtn.textContent = '📌';
                pinBtn.addEventListener('click', (e) => {
                    e.stopPropagation();
                    togglePin(handle.name);
                    pinBtn.classList.toggle('pinned', isPinned(handle.name));
                    pinBtn.title = isPinned(handle.name) ? 'Unpin from header' : 'Pin to header';
                    renderPinnedHeader();
                });

                // File link
                const a = document.createElement('a');
                a.href = '#';
                a.textContent = handle.name;
                a.dataset.filename = handle.name;

                a.addEventListener('click', async (e) => {
                    e.preventDefault();

                    const file = await handle.getFile();
                    const text = await file.text();
                    editor.value = text;
                    showEditor();

                    if (inputAbortController) inputAbortController.abort();
                    inputAbortController = new AbortController();

                    editor.addEventListener(
                        'input',
                        async () => {
                            const writable = await handle.createWritable();
                            await writable.write(editor.value);
                            await writable.close();
                        },
                        { signal: inputAbortController.signal }
                    );
                });

                li.appendChild(pinBtn);
                li.appendChild(a);
                ulFiles.appendChild(li);
            }

            renderPinnedHeader();
        }

        function renderLinks(text) {
            const items = JSON.parse(text);
            const systemDefaultContainer = document.getElementById("systemDefaultContainer");
            systemDefaultContainer.innerHTML = "";

            const staticHeadingContainer = document.getElementById("staticHeadingContainer");
            staticHeadingContainer.innerHTML = ""; // reset before rendering

            const archiveContainer = document.getElementById("archivedContainer");
            archiveContainer.innerHTML = ""; // reset before rendering

            items.forEach(item => {
                const wrapper = document.createElement("h4");

                const a = document.createElement("a");
                a.className = "fillLink";
                a.textContent = item.label || "link";

                if (item.type === "external") {
                    a.href = item.href;
                    a.target = "_blank";
                    systemDefaultContainer.appendChild(a);
                    systemDefaultContainer.appendChild(document.createElement("br"));
                }
                else if (item.type === "action") {
                    a.href = "#";
                    a.dataset.path = item.path;
                    a.dataset.method = item.method;
                    a.addEventListener("click", () => {
                        updatePath(a.dataset.path);
                        updateMethod(a.dataset.method);
                        document.getElementById("input").value = '';
                    });
                    systemDefaultContainer.appendChild(a);
                    systemDefaultContainer.appendChild(document.createElement("br"));
                }
                else if (item.type === "working") {
                    a.href = item.href;
                    a.target = "_blank";
                    staticHeadingContainer.appendChild(a);
                    staticHeadingContainer.appendChild(document.createTextNode(" | "));
                }
                else if (item.type === "archive") {
                    a.href = item.href;
                    a.target = "_blank";
                    archiveContainer.appendChild(a);
                    archiveContainer.appendChild(document.createElement("br"));
                }

            });
        }

        const reconnectOverlay = document.getElementById('reconnectOverlay');
        const reconnectOk = document.getElementById('reconnectOk');

        function showReconnectOverlay() {
            reconnectOverlay.classList.remove('hidden');
        }

        function hideReconnectOverlay() {
            reconnectOverlay.classList.add('hidden');
        }

        reconnectOk?.addEventListener('click', async () => {
            if (!pendingDirHandle) {
                hideReconnectOverlay();
                return;
            }

            try {
                const dir = await restoreOrPickDir();
                if (dir) {
                    hideReconnectOverlay();
                    await renderFileList(dir);
                    pendingDirHandle = null;
                }
                hideReconnectOverlay();
            } catch (e) {
                hideReconnectOverlay();
                console.warn(e);
            }
        });

        function collapseAndDropPanel() {
            const panel = document.getElementById('panel');
            const toggle = document.getElementById('toggle');

            // collapse
            panel.classList.add('collapsed');
            if (toggle) toggle.textContent = '▶';

            // move down using TOP only (no bottom anchoring)
            const margin = 10;
            const panelHeight = panel.offsetHeight;
            const viewportHeight = window.innerHeight;

            panel.style.top = (viewportHeight - panelHeight - margin) + 'px';
        }
